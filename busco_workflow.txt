####BUSCO workflow for SCG completeness estimation ####
#setup env
conda create -n busco -c conda-forge -c bioconda busco=5.7.1
conda activate busco


#clean fasta headers
sed 's/\//_/g' combined_pseudo_nitzschia_transcriptome.fasta > combined_pseudo_nitzschia_transcriptome.clean.fasta


busco -i combined_pseudo_nitzschia_transcriptome.fasta \
    -l stramenopiles_odb10 \
    -m transcriptome \
    -o busco_stramenopiles \
    -f


source busco_scg.R
